MediaPlayerApp.java 
package com.example;

import javafx.application.Application;
import javafx.animation.*;
import javafx.geometry.Insets;
import javafx.geometry.Pos;
import javafx.scene.Scene;
import javafx.scene.control.*;
import javafx.scene.image.Image;
import javafx.scene.image.ImageView;
import javafx.scene.layout.*;
import javafx.scene.media.Media;
import javafx.scene.media.MediaPlayer;
import javafx.stage.FileChooser;
import javafx.stage.Stage;
import javafx.util.Duration;

import java.io.File;

public class MediaPlayerApp extends Application {

    private MediaPlayer player;
    private Timeline pulseTimeline;

    private final Label trackLabel = new Label("Track: (not selected)");
    private final Label statusLabel = new Label("Status: STOPPED");
    private final Label timeLabel = new Label("00:00 / 00:00");

    private final ToggleButton playPauseBtn = new ToggleButton("Play");
    private final ToggleButton muteBtn = new ToggleButton("Mute");
    private final Button stopBtn = new Button("Stop");

    private final Slider volumeSlider = new Slider(0, 1, 0.7);
    private final Slider progressSlider = new Slider();

    private final CheckBox repeatCheck = new CheckBox("Repeat");
    private final CheckBox shuffleCheck = new CheckBox("Shuffle");

    private final RadioButton rbNormal = new RadioButton("Normal");
    private final RadioButton rbBass = new RadioButton("Bass");
    private final RadioButton rbSoft = new RadioButton("Soft");

    @Override
    public void start(Stage stage) {

        /* ===== TOP ===== */
        Label title = new Label("JavaFX Media Player");
        title.setStyle("-fx-font-size: 22; -fx-font-weight: bold; -fx-text-fill: white;");

        Button openBtn = new Button("Open audio");
        openBtn.setOnAction(e -> openFile(stage));

        HBox top = new HBox(15, title, openBtn);
        top.setAlignment(Pos.CENTER_LEFT);
        top.setPadding(new Insets(10));
        top.setStyle("-fx-background-color: rgba(0,0,0,0.45);");

        /* ===== CENTER ===== */
        trackLabel.setStyle("-fx-text-fill: white;");
        statusLabel.setStyle("-fx-text-fill: white;");
        timeLabel.setStyle("-fx-text-fill: white;");

        VBox center = new VBox(10, trackLabel, statusLabel, progressSlider, timeLabel);
        center.setAlignment(Pos.CENTER);
        center.setPadding(new Insets(15));
        center.setStyle(
                "-fx-background-color: rgba(0,0,0,0.35);" +
                        "-fx-background-radius: 15;"
        );

        /* ===== BOTTOM ===== */
        playPauseBtn.setDisable(true);
        stopBtn.setDisable(true);
        muteBtn.setDisable(true);

        playPauseBtn.setOnAction(e -> onPlayPause());
        stopBtn.setOnAction(e -> onStop());
        muteBtn.setOnAction(e -> onMute());

        volumeSlider.setShowTickMarks(true);
        volumeSlider.setShowTickLabels(true);

        HBox bottom = new HBox(
                10,
                playPauseBtn,
                stopBtn,
                muteBtn,
                new Label("Volume"),
                volumeSlider
        );
        bottom.setAlignment(Pos.CENTER);
        bottom.setPadding(new Insets(10));
        bottom.setStyle("-fx-background-color: rgba(0,0,0,0.45);");

        bottom.getChildren().forEach(n -> {
            if (n instanceof Labeled) {
                ((Labeled) n).setStyle("-fx-text-fill: white;");
            }
        });

        /* ===== RIGHT ===== */
        ToggleGroup group = new ToggleGroup();
        rbNormal.setToggleGroup(group);
        rbBass.setToggleGroup(group);
        rbSoft.setToggleGroup(group);
        rbNormal.setSelected(true);

        VBox right = new VBox(
                10,
                new Label("Settings"),
                repeatCheck,
                shuffleCheck,
                new Separator(),
                new Label("Sound mode"),
                rbNormal, rbBass, rbSoft
        );
        right.setPadding(new Insets(12));
        right.setPrefWidth(180);
        right.setStyle("-fx-background-color: rgba(0,0,0,0.45);");

        right.getChildren().forEach(n -> {
            if (n instanceof Labeled) {
                ((Labeled) n).setStyle("-fx-text-fill: white;");
            }
        });

        /* ===== UI LAYOUT ===== */
        BorderPane ui = new BorderPane();
        ui.setTop(top);
        ui.setCenter(center);
        ui.setBottom(bottom);
        ui.setRight(right);
        ui.setPadding(new Insets(10));

        /* ===== BACKGROUND IMAGE ===== */
        String imagePath = "C:/img/muzyka-music-naushniki-serdce.jpg";
        Image bgImage = new Image(new File(imagePath).toURI().toString());

        ImageView bgView = new ImageView(bgImage);
        bgView.setFitWidth(900);
        bgView.setFitHeight(520);
        bgView.setPreserveRatio(false);

        StackPane root = new StackPane(bgView, ui);

        /* ===== FADE IN ANIMATION ===== */
        FadeTransition fade = new FadeTransition(Duration.seconds(1.5), root);
        fade.setFromValue(0);
        fade.setToValue(1);
        fade.play();

        /* ===== PROGRESS PULSE ===== */
        pulseTimeline = new Timeline(
                new KeyFrame(Duration.ZERO,
                        new KeyValue(progressSlider.opacityProperty(), 1)
                ),
                new KeyFrame(Duration.seconds(1),
                        new KeyValue(progressSlider.opacityProperty(), 0.6)
                )
        );
        pulseTimeline.setAutoReverse(true);
        pulseTimeline.setCycleCount(Animation.INDEFINITE);

        stage.setScene(new Scene(root, 900, 520));
        stage.setTitle("Study Media Player");
        stage.show();
    }

    /* ===== OPEN FILE ===== */
    private void openFile(Stage stage) {
        FileChooser chooser = new FileChooser();
        chooser.setTitle("Select audio file");
        chooser.getExtensionFilters().add(
                new FileChooser.ExtensionFilter("Audio", "*.mp3", "*.wav", "*.m4a")
        );

        File file = chooser.showOpenDialog(stage);
        if (file == null) return;

        trackLabel.setText("Track: " + file.getName());

        if (player != null) {
            player.stop();
            player.dispose();
        }

        Media media = new Media(file.toURI().toString());
        player = new MediaPlayer(media);

        player.volumeProperty().bind(volumeSlider.valueProperty());

        player.currentTimeProperty().addListener((o, oldT, newT) -> {
            if (!progressSlider.isValueChanging()) {
                progressSlider.setValue(newT.toSeconds());
            }
            updateTime();
        });

        player.setOnReady(() -> {
            progressSlider.setMax(player.getTotalDuration().toSeconds());
            updateTime();
            statusLabel.setText("Status: READY");
        });

        progressSlider.setOnMouseReleased(e ->
                player.seek(Duration.seconds(progressSlider.getValue()))
        );

        player.setOnEndOfMedia(() -> {
            if (repeatCheck.isSelected()) {
                player.seek(Duration.ZERO);
                player.play();
            } else {
                playPauseBtn.setSelected(false);
                playPauseBtn.setText("Play");
                pulseTimeline.stop();
                progressSlider.setOpacity(1);
                statusLabel.setText("Status: FINISHED");
            }
        });

        playPauseBtn.setDisable(false);
        stopBtn.setDisable(false);
        muteBtn.setDisable(false);
    }

    /* ===== PLAY / PAUSE ===== */
    private void onPlayPause() {
        if (player == null) return;

        if (playPauseBtn.isSelected()) {
            player.play();
            playPauseBtn.setText("Pause");
            statusLabel.setText("Status: PLAYING");
            pulseTimeline.play();

            ScaleTransition scale = new ScaleTransition(Duration.millis(200), playPauseBtn);
            scale.setFromX(1);
            scale.setToX(1.15);
            scale.setFromY(1);
            scale.setToY(1.15);
            scale.setAutoReverse(true);
            scale.setCycleCount(2);
            scale.play();

        } else {
            player.pause();
            playPauseBtn.setText("Play");
            pulseTimeline.stop();
            progressSlider.setOpacity(1);
            statusLabel.setText("Status: PAUSED");
        }
    }

    /* ===== STOP ===== */
    private void onStop() {
        if (player == null) return;
        player.stop();
        playPauseBtn.setSelected(false);
        playPauseBtn.setText("Play");
        pulseTimeline.stop();
        progressSlider.setOpacity(1);
        statusLabel.setText("Status: STOPPED");
    }

    /* ===== MUTE ===== */
    private void onMute() {
        if (player == null) return;
        player.setMute(muteBtn.isSelected());
        muteBtn.setText(muteBtn.isSelected() ? "Unmute" : "Mute");

        FadeTransition f = new FadeTransition(Duration.millis(300), muteBtn);
        f.setFromValue(0.4);
        f.setToValue(1);
        f.play();
    }

    /* ===== TIME ===== */
    private void updateTime() {
        if (player == null) return;
        Duration c = player.getCurrentTime();
        Duration t = player.getTotalDuration();
        timeLabel.setText(format(c) + " / " + format(t));
    }

    private String format(Duration d) {
        int s = (int) d.toSeconds();
        return String.format("%02d:%02d", s / 60, s % 60);
    }

    public static void main(String[] args) {
        launch(args);
    }
}
